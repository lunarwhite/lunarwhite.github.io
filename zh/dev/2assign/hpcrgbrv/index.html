#include"mpi.h"
#include<stdio.h>
#include<math.h>
void main(int argc, char *argv[]){
    // 相关变量声明
    int numprocs;
    int procnum;
    int namelen;
    char pro_name[MPI_MAX_PROCESSOR_NAME];
        
    // 初始化MPI环境
    MPI_Init(&argc, &argv);

    // MPI程序体
    MPI_Comm_rank(MPI_COMM_WORLD, &procnum);
    MPI_Comm_size(MPI_COMM_WORLD, &numprocs);
    MPI_Get_processor_name(pro_name, &namelen);
        
    printf("Hello World! Hello HPC! Processor %d of %d on %s \n", procnum, numprocs, pro_name);

    // 退出MPI环境
    MPI_Finalize();
}

#include <stdio.h>
#include "mpi.h"
int main(int argc, char *argv[]){
    int rank, value, size;

    MPI_Status status;
    MPI_Init(&argc, &argv);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    do {
        if (rank==0) {
            fprintf(stderr, "\n\nPlease give new value = ");
            scanf("%d",&value);
            fprintf(stderr, "\n%d read <-from ( %d )",rank,value);
            if (size>1) {
                MPI_Send(&value, 1, MPI_INT, rank+1, 0, MPI_COMM_WORLD);
                fprintf(stderr, "\n%d send ( %d ) to-> %d", rank,value,rank+1);
            }
        } else {
            MPI_Recv(&value, 1, MPI_INT, rank-1, 0, MPI_COMM_WORLD, &status);
            fprintf(stderr, "\n%d receive( %d )<-from %d",rank, value, rank-1);
            if (rank<size-1) {
                MPI_Send(&value, 1, MPI_INT, rank+1, 0, MPI_COMM_WORLD);
                fprintf(stderr, "\n%d send ( %d ) to-> %d", rank, value, rank+1);
            }
        }
        MPI_Barrier(MPI_COMM_WORLD);
    } while (value>0);

    MPI_Finalize();
    printf("\n");
}

    MPI_Init( &argc, &argv );

    MPI_Comm_size( MPI_COMM_WORLD, &size );

    if (myid == 0) {
        for (i=0; i<10*(size-1); i++) {
            MPI_Recv( buf, 1, MPI_INT, MPI_ANY_SOURCE, MPI_ANY_TAG, MPI_COMM_WORLD, &status );
            fprintf(stderr, "Msg=%d from Process %d with tag %d\n", buf[0], status.MPI_SOURCE, status.MPI_TAG );
        }
    } else {
        for (i=0; i<10; i++){
            buf[0]=10*myid+i;
            MPI_Send( buf, 1, MPI_INT, 0, i, MPI_COMM_WORLD );
        }
    }
    MPI_Finalize();
}

2.1 一对多集合通信函数MPI_Bcast

  • int MPI_Bcast(void *buf, int count, MPI_Datatype datatype, int root, MPI_Comm comm)
  • 在一个通信域内，以其中一个进程为根进程，向其他所有进程（包括该进程自身）发送数据。所有进程都使用同一个通信域comm和根标示root。其他进程指定的通信元素个数、数据类型必须与根进程指定的通信元素个数、数据类型保持一致
  • 一对多集合通信函数还有MPI_Scatter

#include "mpi.h"
#include <stdio.h>
#include <stdlib.h>

int main(int argc, char *argv[]){
    int rank, value=1;
    MPI_Init(&argc, &argv);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    
    while(value>0){
        if (0==rank){
            printf("Please input a positive value: 0 to exit\n");
            scanf("%d", &value);
        }
        
        // 1是通信元素个数，0是root进程
        MPI_Bcast(&value, 1, MPI_INT, 0, MPI_COMM_WORLD);
        printf("Process %d got value %d\n", rank, value);
    }
    
    MPI_Finalize();
    return 0;
}

#include <stdlib.h>
#include <omp.h>
#include <stdio.h>
#define NUM_THREADS 8 //设置线程数目为8

int main(){
    //--串行区，由一个初始线程执行
    static long num_steps=1000000000;
    double step;
    int i;
    double x,pi,sum=0.0;

    step=1.0/(double) num_steps;

    //调用运行时库函数，为后面的并行区设置线程数目
    omp_set_num_threads(NUM_THREADS);

    //编译指导语句，创建一个并行区。在该并行区中多线程执行
    #pragma omp parallel for reduction(+:sum) private(x)

    //--并行区，该区域由8个线程并行执行
    for(i=0;i<num_steps;i++){
        x=(i+0.5)*step;
        sum+=4.0/(1.0+x*x);
    }

    //--串行区，只有一个主线程执行
    pi=step*sum;
    printf("Pi=%21.20f(%ld steps)\n",pi,num_steps);
    return 0;
}

#include <stdlib.h>
#include <omp.h>
#include <stdio.h>
#define NUM_THREADS 8 //设置线程数目为8
#define N 100

    int main(){
    //--串行区，由一个初始线程执行
    int i;
    int x[N];
    int y[N];
    int z[N];

    //初始化x[i] y[i]
    for(i=0; i<N; i++){
        x[i]=i;
        y[i]=i*i;
    }

    //调用运行时库函数，为后面的并行区设置线程数目
    omp_set_num_threads(NUM_THREADS);

    //编译指导语句，创建一个并行区。在该并行区中多线程执行
    #pragma omp parallel for
    for(i=0; i<N; i++){ //--并行区，该区域由8个线程并行执行
        z[i] = x[i]+y[i];
    }

    //--串行区，只有一个主线程执行
    for(i=0; i<N; i++){
        printf("z[%d]=x[%d]+y[%d]=%d\n", i, i, i, z[i]);
    }
    return 0;
}

#include <stdlib.h>
#include <stdio.h>
#include <omp.h> 
#define NUM_THREADS 8 //设置线程数目为8

int main(){
    //--串行区
    int a1=0;
    int a2=0;
    int a3=0;
    int i=1;

    //设置线程数目
    omp_set_num_threads(NUM_THREADS);

    //无归约操作求解1-100的和，计算结果不确定
    #pragma omp parallel for private(i)
    for(i=1;i<101;i++){
        a1=a1+i;
    }

    //归约操作求解1-100的和，计算结果正确
    #pragma omp parallel for private(i) reduction(+:a)
    for(i=1;i<101;i++){
        a2=a2+i;
    }

    //串行操作
    for(i=1;i<101;i++){
        a3=a3+i;
    }

    //显示结果
    printf("并行无归约操作\t a1=%d\n",a1);
    printf("并行归约操作\t a2=%d\n",a2);
    printf("串行操作\t a3=%d\n",a3);

    return 0;
}

#include <stdlib.h>
#include <stdio.h>
#include <omp.h> 
#define NUM_THREADS 3 //设置线程数目为8

int alpha(){
    return 100;
}

int beta(){
    return 600;
}

int delta(){
    return 400;
}

int plus(int v,int w){
    return v+w;
}

int multiply(int x,int y){
    return x*y;
}

int main(){
    //设置线程数目
    omp_set_num_threads(NUM_THREADS);

    //初始化
    int v, w, y, x, z;

    //首先并行处理得到v，w，然后并行处理得到x，y，
    //最后打印出z
    //sections与sections之间串行执行

    #pragma omp parallel
    {
        #pragma omp sections
        {
            #pragma omp section
            v = alpha(); 

            #pragma omp section
            w = beta();
        } 
        #pragma omp sections
        {
        #pragma omp section
        x = plus(v,w);   
        
        #pragma omp section
        y = delta();    
        }   
    }

    z = multiply(x, y);

    printf("v=alpha()=%d\n", v);
    printf("w=beta()=%d\n", w);
    printf("y=delta()=%d\n", y);
    printf("x=plus(v, w)=%d\n", x);
    printf("z=multiply(x, y)=%d\n", z);
        
    return 0;
}

__global__ void add_gpu(float *a, float *b, int NN) {

        a[idx] += b[idx];
}

    int i;
    for (i = 0; i < NN; ++i)
        a[i] = rand() / (double)(RAND_MAX / 100);

int main() {
    float *a, *b;
    float *d_a, *d_b;
    int size = N * sizeof(int);

    cudaMalloc((void**)&d_a, size);
    cudaMalloc((void**)&d_b, size);

    a = (float *)malloc(size); random_ints(a, N);
    b = (float *)malloc(size); random_ints(b, N);
    
    for (int i = 0; i < N; i++) {
        printf("a[%d] = %f", i, a[i]);
        printf("\tb[%d] = %f\n", i, b[i]);
    }

    // copy inputs to device
    cudaMemcpy(d_a, a, size, cudaMemcpyHostToDevice);
    cudaMemcpy(d_b, b, size, cudaMemcpyHostToDevice);

    }

    // cleanup
    free(a);
    free(b);
    cudaFree(d_a);
    cudaFree(d_b);

    return 0;
}

#include "cuda_runtime.h"
#include "device_launch_parameters.h"
#include <stdio.h>
#define Row  126
#define Col 250

__global__ void matrix_mul_gpu(int *M, int* N, int* P, int width) {
    int i = threadIdx.x + blockDim.x * blockIdx.x;
    int j = threadIdx.y + blockDim.y * blockIdx.y;

    int sum = 0;
    for(int k=0;k<width;k++) {
        int a = M[j*width+k];
        int b = N[k*width+i];
        sum += a*b;
    }
    P[j*width+i] = sum;
}

int main(){
    int *d_dataA, *d_dataB, *d_dataC;
    
    // alloc for host
    int *A = (int *)malloc(sizeof(int) * Row * Col);
    int *B = (int *)malloc(sizeof(int) * Row * Col);
    int *C = (int *)malloc(sizeof(int) * Row * Col);

    // alloc for device
    cudaMalloc((void**)&d_dataA, sizeof(int) *Row*Col);
    cudaMalloc((void**)&d_dataB, sizeof(int) *Row*Col);
    cudaMalloc((void**)&d_dataC, sizeof(int) *Row*Col);

    for (int i = 0; i < Row*Col; i++) {
        A[i] = 90;
        B[i] = 10;
    }
    
    // copy inputs to device
    cudaMemcpy(d_dataA, A, sizeof(int) * Row * Col, cudaMemcpyHostToDevice);
    cudaMemcpy(d_dataB, B, sizeof(int) * Row * Col, cudaMemcpyHostToDevice);

    dim3 threadPerBlock(16, 16);
    dim3 blockNumber((Col+threadPerBlock.x-1)/ threadPerBlock.x, (Row+threadPerBlock.y-1)/ threadPerBlock.y );

    printf("Block(%d,%d)   Grid(%d,%d).\n", threadPerBlock.x, threadPerBlock.y, blockNumber.x, blockNumber.y);
    matrix_mul_gpu << <blockNumber, threadPerBlock >> > (d_dataA, d_dataB, d_dataC, Col);
    
    // copy result back to host
    cudaMemcpy(C, d_dataC, sizeof(int) * Row * Col, cudaMemcpyDeviceToHost);

    // cleanup
    free(A);
    free(B);
    free(C);
    cudaFree(d_dataA);
    cudaFree(d_dataB);
    cudaFree(d_dataC);

    return 0;
}
